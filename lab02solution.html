1. Area & Perimeter of Rectangle, Area & Circumference of Circle.

#include <stdio.h>
int main() {
    float length, breadth, radius;
    float rect_area, rect_perimeter, circ_area, circumference;
    const float PI = 3.14159;

    printf("Enter length of rectangle: ");
    scanf("%f", &length);
    printf("Enter breadth of rectangle: ");
    scanf("%f", &breadth);
    printf("Enter radius of circle: ");
    scanf("%f", &radius);

    rect_area = length * breadth;
    rect_perimeter = 2 * (length + breadth);
    circ_area = PI * radius * radius;
    circumference = 2 * PI * radius;

    printf("Area of rectangle: %.2f\n", rect_area);
    printf("Perimeter of rectangle: %.2f\n", rect_perimeter);
    printf("Area of circle: %.2f\n", circ_area);
    printf("Circumference of circle: %.2f\n", circumference);

    return 0;
}

3. Calculate the sum of digits for a five-digit number.

#include <stdio.h>
int main() {
    int n, sum = 0, remainder;
    printf("Enter a five-digit number: ");
    scanf("%d", &n);
    
    // Make a copy to preserve original number if needed
    int temp = n; 
    
    while (temp != 0) {
        remainder = temp % 10;
        sum = sum + remainder;
        temp = temp / 10;
    }
    
    printf("Sum of digits: %d\n", sum);
    return 0;
}

4. Reverse a five-digit number.

#include <stdio.h>
int main() {
    int n, reversed_number = 0, remainder;
    printf("Enter a five-digit number: ");
    scanf("%d", &n);
    
    while (n != 0) {
        remainder = n % 10;
        reversed_number = reversed_number * 10 + remainder;
        n /= 10;
    }
    
    printf("Reversed number: %d\n", reversed_number);
    return 0;
}

5. Sum of the first and last digit of a four-digit number.

#include <stdio.h>
int main() {
    int n, first_digit, last_digit, sum;
    printf("Enter a four-digit number: ");
    scanf("%d", &n);
    
    last_digit = n % 10;
    first_digit = n;
    
    while (first_digit >= 10) {
        first_digit = first_digit / 10;
    }
    
    sum = first_digit + last_digit;
    printf("Sum of first and last digit: %d\n", sum);
    return 0;
}

6. Find the total number of illiterate men and women.

#include <stdio.h>
int main() {
    long total_pop = 80000;
    long total_men, total_women;
    long total_literacy, literate_men, literate_women;
    long illiterate_men, illiterate_women;

    total_men = (52L * total_pop) / 100L;
    total_women = total_pop - total_men;
    
    total_literacy = (48L * total_pop) / 100L;
    literate_men = (35L * total_pop) / 100L;
    literate_women = total_literacy - literate_men;

    illiterate_men = total_men - literate_men;
    illiterate_women = total_women - literate_women;
    
    printf("Illiterate Men: %ld\n", illiterate_men);
    printf("Illiterate Women: %ld\n", illiterate_women);

    return 0;
}

7. Calculate currency notes for a cashier.

#include <stdio.h>
int main() {
    int amount_in_hundreds, total_amount;

    printf("Enter amount to be withdrawn (in hundreds): ");
    scanf("%d", &amount_in_hundreds);

    total_amount = amount_in_hundreds * 100;

9. Print a new number by adding one to each of its digits.

#include <stdio.h>
int main() {
    int number, new_number = 0, remainder, place_value = 1;
    printf("Enter a five-digit number: ");
    scanf("%d", &number);

    while (number > 0) {
        remainder = number % 10;
        int new_digit = (remainder + 1) % 10;
        new_number = new_number + new_digit * place_value;
        place_value *= 10;
        number /= 10;
    }

    printf("New number: %d\n", new_number);
    return 0;
}

10. Calculate insurance premium based on given rules.

#include <string.h>

    scanf("%d", &age);
    printf("Enter location (city/village): ");

    scanf("%s", sex);

    if (strcmp(health, "excellent") == 0 && (age >= 25 && age <= 35) && strcmp(location, "city") == 0) {
        if (strcmp(sex, "male") == 0) {
            printf("Status: Insured\n");
            printf("Premium Rate: Rs. 4 per thousand\n");
            printf("Maximum Policy Amount: Rs. 2,00,000\n");
        } else if (strcmp(sex, "female") == 0) {
            printf("Status: Insured\n");
            printf("Premium Rate: Rs. 3 per thousand\n");
            printf("Maximum Policy Amount: Rs. 1,00,000\n");
        } else {
            printf("Status: Not Insured\n");
        }
    } else if (strcmp(health, "poor") == 0 && (age >= 25 && age <= 35) && strcmp(location, "village") == 0 && strcmp(sex, "male") == 0) {
        printf("Status: Insured\n");
        printf("Premium Rate: Rs. 6 per thousand\n");
        printf("Maximum Policy Amount: Rs. 10,000\n");
    } else {
        printf("Status: Not Insured\n");
    }

    return 0;
}
